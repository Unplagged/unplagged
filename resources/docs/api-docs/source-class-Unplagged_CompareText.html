 21: /**
 22:  * This class can be used to find similarities in text corpora.
 23:  * 
 24:  * It is based on the work done by {@link http://de.vroniplag.wikia.com/ VroniPlag} user "Marcusb". The original 
 25:  * javascript sources can be found under {@link http://de.vroniplag.wikia.com/wiki/MediaWiki:Fragcolors/code.js}(last
 26:  * accessed Nov. 9th 2012).
 27:  */
 28: class Unplagged_CompareText{
 29: 
 30:   private $wordSimilarityMinimum;
 31:   private $markerCount;
 32:   private $leftSide;
 33:   private $rightSide;
 34: 
 35:   /**
 36:    * 
 37:    * @param int $wordSimilarityMinimum The count of words which are minimally needed for the match to be marked as a 
 38:    * similarity.
 39:    * @param int $markerCount The count of different markers that should be used to denote the similarities. 
 40:    */
 41:   public function __construct($wordSimilarityMinimum = 3, $markerCount = 9){
 42:     $this->wordSimilarityMinimum = $wordSimilarityMinimum;
 43:     $this->markerCount = $markerCount;
 44:   }
 45: 
 46:   /**
 47:    * 
 48:    * @param array $linesLeft
 49:    * @param array $linesRight
 50:    * @param int $minlength
 51:    * @return type
 52:    */
 53:   public function compare(array $listLeft, array $listRight){
 54:     $wordsLeft = $this->buildSingleWordsArray($listLeft);
 55:     $wordsRight = $this->buildSingleWordsArray($listRight);
 56: 
 57:     $documents = array($wordsLeft, $wordsRight);
 58: 
 59:     $comparedTexts = $this->compareText($documents);
 60: 
 61:     return $this->getMarkedTexts($comparedTexts, $wordsLeft, $wordsRight, $listLeft, $listRight);
 62:   }

 64:   /**
 65:    * Builds a continuously indexed array of all "words"(including punctuation marks) from the given array of lines.
 66:    * 
 67:    * @param array $lines
 68:    * @return array
 69:    */
 70:   private function buildSingleWordsArray(array $lines){
 71:     $concatenatedLines = implode(' ', $lines);
 72: 
 73:     return array_values(explode(' ', $concatenatedLines));
 74:   }

 76:   /**
 77:    * 
 78:    * @param array $documents
 79:    * @return array
 80:    */
 81:   private function compareText($documents){
 82:     $documents_len = array();
 83: 
 84:     $final_match_list = array();
 85:     $match_table = array();
 86:     $docs = sizeof($documents);
 87: 
 88:     for($doc_idx = 0; $doc_idx < $docs; $doc_idx++) {
 89:       $doc = $documents[$doc_idx];
 90:       $tokens = sizeof($doc) - $this->wordSimilarityMinimum + 1;
 91:       $doc_len = sizeof($doc);
 92:       $documents_len[$doc_idx] = $doc_len;
 93: 
 94:       // If the word is smaller than the min_run_length, do not analyse it
 95:       if($tokens <= 0){
 96:         continue;
 97:       }
 98: 
 99:       $min_token_idx = 0;
100: 
101:       for($token_idx = 0; $token_idx < $tokens; $token_idx++) {
102:         $match = array_slice($doc, $token_idx, $this->wordSimilarityMinimum);
103:         $match_loc = array($doc_idx, $token_idx);
104:         $match_tag = implode(" ", $match);
105: 
106:         if(array_key_exists($match_tag, $match_table)){
107:           if($token_idx >= $min_token_idx){
108:             $best_match = array($doc_idx, $token_idx, -1, 0, 0);
109:             $matches = $match_table[$match_tag];
110:             $nr_matches = sizeof($matches);
111:             for($idx = 0; $idx < $nr_matches; $idx++) {
112:               $match_peer = $matches[$idx];
113:               $peer_doc_idx = $match_peer[0];
114:               $peer_doc = $documents[$peer_doc_idx];
115:               $peer_token_idx = $match_peer[1] + $this->wordSimilarityMinimum;
116:               $peer_len = $documents_len[$peer_doc_idx];
117:               $our_token_idx = $token_idx + $this->wordSimilarityMinimum;
118: 
119:               if($peer_doc_idx == $doc_idx){
120:                 continue;
121:               }
122: 
123:               while($peer_token_idx < $peer_len
124:               && $our_token_idx < $doc_len
125:               && $peer_doc[$peer_token_idx] == $doc[$our_token_idx]){
126:                 $peer_token_idx++;
127:                 $our_token_idx++;
128:               }
129: 
130:               $len = $our_token_idx - $token_idx;
131:               if($len > $best_match[4]){
132:                 $best_match[2] = $match_peer[0];
133:                 $best_match[3] = $match_peer[1];
134:                 $best_match[4] = $len;
135:               }
136:             }
137: 
138:             if($best_match[2] != -1){
139:               array_push($final_match_list, $best_match);
140:               $min_token_idx = $token_idx + $best_match[4];
141:             }
142:           }
143:           array_push($match_table[$match_tag], $match_loc);
144:         }else{
145:           $match_table[$match_tag] = array($match_loc);
146:         }
147:       }
148:     }
149:     return $final_match_list;
150:   }
151: 
152:   /**
153:    * 
154:    * @param type $comparedTexts
155:    * @param type $wordsLeft
156:    * @param type $wordsRight
157:    * @param type $listLeft
158:    * @param type $listRight
159:    * @return type
160:    */
161:   private function getMarkedTexts($comparedTexts, $wordsLeft, $wordsRight, &$listLeft, &$listRight){
162: 
163:     for($i = 0; $i < sizeof($comparedTexts); $i++) {
164:       $colorNumber = $i % $this->markerCount;
165:       $res = $comparedTexts[$i];
166:       $wordsLeft[$res[3]] = '<span class="fragmark fragmark-' . $colorNumber . '">' . $wordsLeft[$res[3]];
167:       $wordsLeft[$res[3] + $res[4] - 1] = $wordsLeft[$res[3] + $res[4] - 1] . '</span>';
168: 
169:       $wordsRight[$res[1]] = '<span class="fragmark fragmark-' . $colorNumber . '">' . $wordsRight[$res[1]];
170:       $wordsRight[$res[1] + $res[4] - 1] = $wordsRight[$res[1] + $res[4] - 1] . '</span>';
171:     }
172: 
173:     // At this point the text is exploded word by word, now the line breaks have to be added again, as in the original document
174:     $resultLeft = $this->addLinebreaks($listLeft, $wordsLeft);
175:     $resultRight = $this->addLinebreaks($listRight, $wordsRight);
176: 
177:     return array('left'=>$resultLeft, 'right'=>$resultRight);
178:   }
179: 
180:   /**
181:    * Adds line numbers and linebreaks back to the result.
182:    * 
183:    * @param array $originalLines
184:    * @param array $originalWords
185:    * @return array An array containing the original lines with html marks for the similarities.
186:    */
187:   private function addLinebreaks(array $originalLines, array $originalWords){
188:     $markedLines = array();
189:     //the start index of the next line
190:     $offset = 0;
191: 
192:     $openInNextLine = '';
193:     foreach($originalLines as $lineNumber=> $lineContent) {
194:       $wordsInLine = count(explode(" ", $lineContent));
195: 
196:       $words = array();
197:       for($i = $offset; $i < $offset + $wordsInLine; $i++) {
198:         $words[] = $originalWords[$i];
199:       }
200: 
201:       $lineContent = $openInNextLine . implode(" ", $words);
202: 
203:       // close opened fragmark tags and determine which tags were closed that have to be opened in next line
204:       $closeTagsResult = $this->addClosingTags($lineContent);
205:       $openInNextLine = $closeTagsResult[1];
206: 
207:       $markedLines[$lineNumber] = $closeTagsResult[0];
208:       $offset += $wordsInLine;
209:     }
210: 
211:     return $markedLines;
212:   }

214:   /**
215:    * Closes all open html tags at the end of the given line. The closed tags need to be reopened on the next line, in
216:    * order to have valid html. This is necessary if for example the lines are displayed in a list tag, so that no invalid
217:    * html occurs.
218:    *
219:    * Based on work by Milian <mail@mili.de>.
220:    * 
221:    * @param string $line
222:    * @return array An array containing the line with appended closing tags if necessary and the tags to be 
223:    * opened on the next line.
224:    */
225:   private function addClosingTags($line){
226:     $openingTags = $this->getOpeningHtmlTags($line);
227:     $closingTags = $this->getClosingHtmlTags($line);
228: 
229:     $numberOfOpenedTags = count($openingTags);
230: 
231:     //store the tags that need to be reopened on the next line
232:     $openInNextLine = '';
233:     if(count($closingTags) !== $numberOfOpenedTags){
234:       $openingTags = array_reverse($openingTags);
235: 
236:       $openedTagsWithAttributes = $this->getOpeningHtmlTags($line, true);
237:       //close tags
238:       for($i = 0; $i < $numberOfOpenedTags; $i++) {
239: 
240:         if(!in_array($openingTags[$i], $closingTags)){
241:           //no matching tag was found, so it spans over to the next line 
242:           //-> close here, reopen on the next line with same attributes
243:           $openInNextLine .= '<' . $openedTagsWithAttributes[$i] . '>';
244:           $line .= '</span>';
245:         }else{
246:           //we had a pair of tags, so we need to remove the tag, because it can't close any other tag anymore
247:           unset($closingTags[array_search($openingTags[$i], $closingTags)]);
248:         }
249:       }
250:     }
251: 
252:     return array($line, $openInNextLine);
253:   }
254: 
255:   /**
256:    * @param string $line A line of text that should be searched for opening html tags.
257:    * @param bool $withAttributes Indicating whether the tags should be with attributes or just the name.
258:    * @return array
259:    */
260:   private function getOpeningHtmlTags($line, $withAttributes = false){
261:     $tags = array();
262: 
263:     if($withAttributes){
264:       preg_match_all('#<(span(?: .*))?(?<![/|/ ])>#iU', $line, $tags);
265:     }else{
266:       preg_match_all('#<([a-z]+)(?: .*)?(?<![/|/ ])>#iU', $line, $tags);
267:     }
268: 
269:     return $tags[1];
270:   }
271: 
272:   /**
273:    * @param string $line A line of text that should be searched for closing tags.
274:    * @return array An array containing all closing tags that were found inside the line.
275:    */
276:   private function getClosingHtmlTags($line){
277:     $closedTags = array();
278:     preg_match_all('#</([a-z]+)>#iU', $line, $closedTags);
279: 
280:     return $closedTags[1];
281:   }
