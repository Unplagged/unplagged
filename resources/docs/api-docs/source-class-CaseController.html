313:             if ($role->getInheritedRoles()->contains($defaultRole)) {
314:               if ($defaultRole->getId() != $inheritedRoleId) {
315:                 //changed role in case
316:                 $role->removeInheritedRole($defaultRole);
317:                 break;
318:               } else {
319:                 // role did not change
320:                 $roleChanged = false;
321:               }

324:           }
