211:     $lines = array();
212:     if($inputType == 'text'){
213:       $lines = explode("\n", $content);
214:     }else if($inputType == 'htmltext'){
215:       $lines = explode("<br />", $content);
216:     }else if($inputType == 'array'){
217:       $lines = $content;
218:     }
219: 
220:     $lineNumbers = array();

222:       $lineNumbers[$key + 1] = $key + 1;
223:     }

228:     $this->lines->filter(function($line) use (&$lineNumbers, &$lines, &$removedLines){
229:           if(in_array($line->getLineNumber(), $lineNumbers)){
230:             $line->setContent($lines[$line->getLineNumber() - 1]);

262:     $plagCount = 0;

264:     // amount of plagiarised lines
265:     // @todo: select only approved fragments here
266:     $plagLines = array();
267:     $em = Zend_Registry::getInstance()->entitymanager;
268:     $query = $em->createQuery('
269:       SELECT f FROM Application_Model_Document_Fragment f 
270:       JOIN f.plag pl
271:       JOIN pl.lineFrom lfrom JOIN lfrom.page pfrom
272:       JOIN pl.lineTo lto JOIN lto.page pto
273:       WHERE :pn >= pfrom.pageNumber AND :pn <= pto.pageNumber AND pfrom.document = :documentId');
274:     $query->setParameter("pn", $this->pageNumber);
275:     $query->setParameter("documentId", $this->document->getId());
276:     $fragments = $query->getResult();
277: 
278:     foreach($fragments as $fragment){
279:       $startPageNumber = $fragment->getPlag()->getLineFrom()->getPage()->getPageNumber();
280:       $startLineNumber = $fragment->getPlag()->getLineFrom()->getLineNumber();
281:       $endPageNumber = $fragment->getPlag()->getLineTo()->getPage()->getPageNumber();
282:       $endLineNumber = $fragment->getPlag()->getLineTo()->getLineNumber();
283: 
284:       $firstLineNumber = $this->lines->first()->getLineNumber();
285:       $lastLineNumber = $this->lines->last()->getLineNumber();

287:       // 1) page number is in between two pages, the whole page is plagiarised
288:       if($startPageNumber < $this->pageNumber && $endPageNumber > $this->pageNumber){
289:         $plagLines = $this->lines;
290: 
291:         // 2) start page number and end page number are on this page
292:       }elseif($startPageNumber == $this->pageNumber && $endPageNumber == $this->pageNumber){
293:         $this->updatePlagLines($startLineNumber, $endLineNumber, $plagLines);
294: 
295:         // 3) page number is page number of start line and end page number is larger
296:       }elseif($startPageNumber == $this->pageNumber && $endPageNumber > $this->pageNumber){
297:         $this->updatePlagLines($startLineNumber, $lastLineNumber, $plagLines);
298: 
299:         // 4) page number is page number somewhere between and end page number is on the same page
300:       }elseif($startPageNumber == $this->pageNumber && $endPageNumber > $this->pageNumber){
301:         $this->updatePlagLines($firstLineNumber, $endLineNumber, $plagLines);
302:       }
303:     }
